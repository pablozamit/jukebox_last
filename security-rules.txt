rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Regla Global para Admins
    // Otorga acceso total si el UID del usuario está en la colección 'admins'
    match /{document=**} {
      allow read, write: if request.auth != null && exists(/databases/$(database)/documents/admins/$(request.auth.uid));
    }

    // 1. Colección de canciones (songs) - Cola Activa
    match /songs/{songId} {
      // Todos pueden leer las canciones para ver la cola y los votos
      allow read: if true;
      
      // Permitir la CREACIÓN si:
      // a) El usuario está logueado.
      // b) El valor inicial de votos es 1.
      allow create: if request.auth != null
                    && request.resource.data.votes == 1;

      // Permitir la ACTUALIZACIÓN de votos SOLO si:
      // a) El usuario está logueado.
      // b) Solo se está modificando 'votes' (incremento de +1) y 'firstVotedAt'.
      allow update: if request.auth != null 
                    && (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['votes', 'firstVotedAt']))
                    && request.resource.data.votes == resource.data.votes + 1;
                    
      // Solo admins pueden borrar (vía regla global)
      allow delete: if false;
    }

    // 2. Catálogo completo (Estructura estática)
    match /catalog/full_list {
      allow read: if true;
      allow write: if false; // Solo admins (regla global)
    }

    // 3. Estado de reproducción (state)
    match /state/nowPlaying {
      allow read: if true;
      allow write: if false; 
    }

    // 4. Usuarios y tokens (users)
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // 5. Comandos (commands)
    match /commands/{commandId} {
      allow read: if false;
      allow write: if false; // Solo admins (regla global)
    }

    // 6. Sugerencias (suggestions)
    match /suggestions/{docId} {
      allow create: if request.auth != null;
      allow read, update, delete: if false;
    }
  }
}
